Backward oracle matching algorithm.

Example

use bio::pattern_matching::bom::BOM;
let text = b"ACGGCTAGGAAAAAGACTGAGGACTGAAAA";
let pattern = b"GAAAA";
let bom = BOM::new(pattern);
let occ: Vec<usize> = bom.find_all(text).collect();
assert_eq!(occ, [8, 25]);
